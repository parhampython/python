import turtle
t=turtle.Turtle()
t.write('start')
t.screen.bgcolor('lightblue')
t.fillcolor('black')
t.begin_fill()
t.fd(100)
t.rt(90)
t.fd(200)
t.rt(90)
t.fd(100)
t.rt(90)
t.fd(200)
t.end_fill()
t.up()
t.goto(80,-50)
t.down()
t.fillcolor('red')
t.begin_fill()
t.circle(25)
t.end_fill()
t.up()
t.goto(80,-100)
t.down()
t.fillcolor('yellow')
t.begin_fill()
t.circle(25)
t.end_fill()
t.up()
t.goto(80,-150)
t.down()
t.fillcolor('green')
t.begin_fill()
t.circle(25)
t.end_fill()
